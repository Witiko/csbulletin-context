\writestatus{loading}{CSTUG bulletin experimental ConTeXt style}%
\unprotect

% Module interface
\definenamespace[csbulletin]
 [type=module,
  name=csbulletin,
  setup=list]

% Fonts
\definebodyfontenvironment[10pt][interlinespace=11dd]
\definebodyfontenvironment[12pt][interlinespace=14pt]
\setupbodyfont[10pt]
\setuppapersize[A5][A5]

% Layout
% \showframe
\setuplayout
  [height=fit,
   topspace=13mm,
   header=0mm,
   headerdistance=0mm,
   footer=\dimexpr210mm-13mm-13mm-492pt\relax,
   footerdistance=0mm,
   lines=42,
   width=fit,
   rightmargin=16mm,
   rightmargindistance=0mm,
   backspace=\dimexpr148mm-27cc-16mm\relax]
\setupindenting[yes,first,15pt]
\setuppagenumbering
  [alternative=doublesided,
   location={footer,right}]

% Section heads
\define[2]\csbulletinsection{{\switchtobodyfont[12pt]#2}}
\define[2]\csbulletinsubsection{{#2}}
\define[2]\csbulletinsubsubsection{{#2}}
\setuphead[section]
 [style=bold,
  number=no,
  command=\csbulletinsection,
  before={\blank[20.5dd]},
  after={\blank[7dd]\blank[none,disable]}]
\setuphead[subsection]
 [style=bold,
  number=no,
  command=\csbulletinsubsection,
  before={\blank[19.5dd]},
  after={\blank[1pt]\blank[none,disable]}]
\setuphead[subsubsection]
 [style=medium,
  number=no,
  command=\csbulletinsubsubsection,
  before={\blank[19.5dd]},
  after={\blank[1pt]\blank[none,disable]}]

% Footnotes
\setupnotation[footnote]
 [alternative=left,
  width=10pt,
  numbercommand=\hskip10pt\high]

% Verbatim text
\setuptyping[before={\blank[1.5pt]},after={\blank[1.5pt]}]

\stopmodule\protect
