\writestatus{loading}{CSTUG bulletin experimental ConTeXt style}%
\unprotect

% Module interface
\definenamespace[csbulletin]
 [type=module,
  name=csbulletin,
  setup=list]

% Fonts
\setupbodyfont[10pt]
\setupinterlinespace[line=11.78pt]
\setuppapersize[A5][A5]

% Layout
% \showframe
\setuplayout
  [height=fit,
   topspace=13mm,
   header=0mm,
   headerdistance=0mm,
   footer=\dimexpr210mm-13mm-13mm-492pt\relax,
   footerdistance=0mm,
   lines=42,
   width=fit,
   rightmargin=16mm,
   rightmargindistance=0mm,
   backspace=\dimexpr148mm-27cc-16mm\relax]
\setupindenting[yes,first,15pt]
\setuppagenumbering
  [alternative=doublesided,
   location={footer,right}]

% Section heads
\define[2]\csbulletinsection{{\setupbodyfont[12pt]#2}}
\define[2]\csbulletinsubsection{{#2}}
\define[2]\csbulletinsubsubsection{{#2}}
\setuphead[section]
 [style=bold,
  number=no,
  command=\csbulletinsection,
  interlinespace=14pt,
  before={\blank[22dd]},
  after={\blank[11dd]\blank[disable]}]
\setuphead[subsection]
 [style=bold,
  number=no,
  command=\csbulletinsubsection,
  before={\blank[22dd]},
  after={\blank[1pt]\blank[disable]}]
\setuphead[subsubsection]
 [style=medium,
  number=no,
  command=\csbulletinsubsubsection,
  before={\blank[22dd]},
  after={\blank[1pt]\blank[disable]}]

\stopmodule\protect
